获取AD用户信息，导出csv文件

Get-ADUser -SearchBase "OU=CorpUser,OU=ROOT,DC=minicontoso,DC=com" -Filter * -Properties * | Select-Object EmployeeID, Description, GivenName, Surname, Department, Title, Office, @{Name='Manager';Expression={$_.Manager.split(",")[0].split("=")[1]}} | Export-Csv -Path D:\CorpUsers.csv -Encoding UTF8
